%%% Блок без заголовка
\newenvironment{block*}[1][structure!10!bg]{%
  \setbeamercolor{block body}{bg=#1}
  \begin{block}{\vspace*{-3ex}}
  }{%
  \end{block}
}

%%% Окружение stripe (полоса)
\newenvironment<>{stripe}[1][structure!10!bg]{%
  \setbeamercolor{block body}{bg=#1}
  \vspace*{-2ex}
  \begin{columns}
    \begin{column}{\paperwidth}
      \begin{block}{\vspace*{-3ex}}
        \hspace*{1em}
        \begin{minipage}{0.94\paperwidth}
        }{%
        \end{minipage}
      \end{block}
    \end{column}
  \end{columns}
}
